\begin{abstract}
\textbf{Primary Result (Theorem~\ref{thm:penroseinitial}):} We prove the \textbf{sharp} Spacetime Penrose Inequality for \textbf{apparent horizons} (outermost MOTS) \textbf{unconditionally}:
\[
M_{\mathrm{ADM}}(g) \ge \sqrt{\frac{A(\Sigma^*)}{16\pi}},
\]
where $\Sigma^*$ is the outermost marginally outer trapped surface in asymptotically flat initial data $(M^3,g,k)$ satisfying the dominant energy condition (DEC). This is the \textbf{main contribution} of this paper. The proof uses the $p \to 1^+$ limit of the p-harmonic level set method (Agostiniani--Mazzieri--Oronzio) combined with the Generalized Jang equation (Han--Khuri). The key technical advance is establishing $[H] \ge 0$ at stable MOTS via the Principal Eigenvalue Theorem for non-self-adjoint operators.

\textbf{Secondary Results (Conditional):} We also prove the inequality for \textbf{general trapped surfaces} under additional hypotheses:
\begin{itemize}
    \item Under Weak Cosmic Censorship (completing Penrose's original 1973 conjecture), or
    \item Under compactness assumptions (C1)--(C3), combined with the time-symmetric restriction $k=0$.
\end{itemize}

\textbf{Critical Open Gap (for $k \neq 0$):} For general trapped surfaces with non-trivial extrinsic curvature $k \neq 0$ and without cosmic censorship, the proof has a \textbf{fundamental gap}: upgrading the integral condition $\int_\Sigma \tr_\Sigma k \, dA \ge 0$ to pointwise $\tr_\Sigma k \ge 0$ remains \textbf{open} due to non-self-adjointness of the MOTS stability operator $L_\Sigma = -\Delta - |A|^2 - \mathrm{Ric}(\nu,\nu) + 2X \cdot \nabla$. This gap is explicitly identified in Remark~\ref{rem:NonSelfAdjointGap}. \textbf{We do not claim to have resolved the general spacetime Penrose inequality for $k \neq 0$ without cosmic censorship.}

\textbf{Fundamental Obstruction (Theorem~\ref{thm:Obstruction}):} We prove that conformal methods \textbf{cannot} simultaneously achieve area preservation and mass reduction for unfavorable jump ($\tr_\Sigma k < 0$). This explains why the outermost MOTS case succeeds unconditionally (stability guarantees favorable jump) while general trapped surfaces require additional hypotheses.
\end{abstract}

\textbf{SECONDARY RESULTS (CONDITIONAL)}

\textbf{For general trapped surfaces} $\Sigma$ (not necessarily outermost), the inequality:
\begin{equation*}
    M_{\mathrm{ADM}} \ge \sqrt{\frac{A(\Sigma)}{16\pi}}

holds \textbf{conditionally} under one of:
\begin{itemize}
    \item Weak Cosmic Censorship (WCC)---completing Penrose's 1973 conjecture
    \item Compactness (C1)--(C3) \textbf{plus} $k=0$ (time-symmetric)
\end{itemize}
