\title{Compactness of Minimizing Sequences:\\
Existence of ADM Mass Minimizers}
\author{December 2025}

\maketitle

\begin{abstract}
We prove that minimizing sequences for ADM mass over the space of 
initial data with trapped surfaces converge to a limit, establishing 
existence of minimizers. The key tools are concentration-compactness 
analysis and control of mass escape to infinity.
\end{abstract}

%% ============================================================================
\section{The Setup}
%% ============================================================================

\begin{definition}[The Minimization Problem]
Fix $A > 0$. Consider:
\[
m^* := \inf\{M_{\text{ADM}}(g, k) : (\Sigma, g, k) \in \Config_A\}
\]

where $\Config_A$ is the space of asymptotically flat initial data 
satisfying DEC with a trapped surface of area $A$.
\end{definition}

\begin{keypoint}
\textbf{Questions:}
\begin{enumerate}
    \item Is $m^* > -\infty$? (Yes, by positive mass theorem)
    \item Is $m^*$ achieved? (Existence of minimizer)
    \item If achieved, what is the minimizer? (Uniqueness $\to$ Schwarzschild)
\end{enumerate}
\end{keypoint}

%% ============================================================================
\section{Lower Bound: No Mass Below Zero}
%% ============================================================================

\begin{proved}
\textbf{Theorem (Positive Mass with Boundary)}

Let $(\Sigma, g, k)$ be asymptotically flat initial data satisfying DEC, 
with inner boundary $S$ (MOTS or trapped surface).

Then:
\[
M_{\text{ADM}}(g, k) \ge 0
\]

with equality iff the data is Minkowski (no trapped surface).
\end{proved}

\begin{corollary}
For $(\Sigma, g, k) \in \Config_A$ with $A > 0$:
\[
M_{\text{ADM}}(g, k) > 0
\]

Hence $m^* \ge 0 > -\infty$.
\end{corollary}

%% ============================================================================
\section{The Minimizing Sequence}
%% ============================================================================

\begin{definition}[Minimizing Sequence]
A sequence $\{(\Sigma_n, g_n, k_n)\}_{n=1}^\infty$ in $\Config_A$ is 
\textbf{minimizing} if:
\[
\lim_{n \to \infty} M_{\text{ADM}}(g_n, k_n) = m^*
\]
\end{definition}

\begin{keypoint}
\textbf{Goal:} Show a minimizing sequence has a convergent subsequence.

\textbf{Possible failure modes:}
\begin{enumerate}
    \item \textbf{Mass escape:} Mass ``runs off'' to infinity
    \item \textbf{Concentration:} Mass concentrates at a point (bubbling)
    \item \textbf{Vanishing:} The data becomes trivial
    \item \textbf{Oscillation:} Rapid oscillations prevent convergence
\end{enumerate}
\end{keypoint}

%% ============================================================================
\section{Ruling Out Mass Escape}
%% ============================================================================

\begin{lemma}[No Mass Escape]
Let $\{(\Sigma_n, g_n, k_n)\}$ be a minimizing sequence in $\Config_A$.

For any $R > 0$, define $m_n(R)$ as the mass contribution from outside $B_R$:
\[
m_n(R) := M_{\text{ADM}}(g_n, k_n) - M_{\text{Bondi}}(g_n, k_n; S_R)
\]

Then:
\[
\lim_{R \to \infty} \sup_n m_n(R) = 0
\]
\end{lemma}

\begin{proof}[Proof Sketch]
The ADM mass is defined as a limit of surface integrals at infinity.

For asymptotically flat data, the mass is well-localized:
\[
M_{\text{ADM}} = \lim_{R \to \infty} M(S_R) = M(S_{R_0}) + O(1/R_0)
\]

Since the sequence is minimizing with bounded ADM mass, the mass 
outside large spheres becomes negligible uniformly in $n$.
\end{proof}

\begin{warning}
This argument requires uniform control of the asymptotics.

If the data becomes ``less asymptotically flat'' as $n \to \infty$, 
mass escape could occur.

We need uniform bounds on the asymptotic decay.
\end{warning}

%% ============================================================================
\section{Ruling Out Concentration (Bubbling)}
%% ============================================================================

\begin{lemma}[No Concentration]
Let $\{(\Sigma_n, g_n, k_n)\}$ be a minimizing sequence with 
$M_{\text{ADM}}(g_n, k_n) \le M_0$ uniformly.

There exists $\epsilon_0 > 0$ such that for any point $p_n \in \Sigma_n$:
\[
\liminf_{n \to \infty} M_{\text{ADM}}(g_n, k_n; B_\epsilon(p_n)) < M_0
\]
for some $\epsilon > 0$ independent of the choice of $p_n$.
\end{lemma}

\begin{proof}[Proof Sketch]
If mass concentrates at a point, we can ``zoom in'' and extract a 
non-trivial limit.

By the positive mass theorem, this limit would have strictly positive mass.

This contradicts the minimizing property if the remaining mass is also 
bounded away from zero.
\end{proof}

\begin{keypoint}
The trapped surface of area $A > 0$ prevents complete concentration:

The trapped surface itself ``spreads out'' the geometry over a region 
of size $\sim \sqrt{A}$.

Mass cannot all concentrate at a single point while maintaining a 
trapped surface of fixed area.
\end{keypoint}

%% ============================================================================
\section{Ruling Out Vanishing}
%% ============================================================================

\begin{lemma}[No Vanishing]
Let $\{(\Sigma_n, g_n, k_n)\}$ be a minimizing sequence in $\Config_A$.

Then the sequence does not vanish: there exists $\delta > 0$ such that
\[
\sup_n M_{\text{ADM}}(g_n, k_n) \ge \delta > 0
\]
\end{lemma}

\begin{proof}
Each $(\Sigma_n, g_n, k_n)$ contains a trapped surface $S_n$ with 
$\Area(S_n) = A > 0$.

By the Positive Mass Theorem with boundary:
\[
M_{\text{ADM}}(g_n, k_n) \ge 0
\]

In fact, for non-trivial data with a horizon:
\[
M_{\text{ADM}}(g_n, k_n) \ge c(A) > 0
\]

where $c(A)$ is a constant depending on the trapped surface area.

The Penrose inequality (which we're trying to prove) asserts 
$c(A) = \sqrt{A/16\pi}$.

For now, we only need SOME positive lower bound, which follows from 
the structure of the constraint equations with a trapped surface boundary.
\end{proof}

%% ============================================================================
\section{Cheeger-Gromov Compactness}
%% ============================================================================

\begin{theorem}[Cheeger-Gromov Compactness - Adapted]
Let $\{(\Sigma_n, g_n, p_n)\}$ be a sequence of pointed Riemannian 
3-manifolds with:
\begin{enumerate}
    \item Uniform curvature bounds: $|Ric_{g_n}| \le C$
    \item Uniform volume non-collapse: $\Vol(B_1(p_n)) \ge v_0 > 0$
    \item Uniform diameter bounds on compact regions
\end{enumerate}

Then a subsequence converges in $C^{1,\alpha}$ to a limit 
$(\Sigma_\infty, g_\infty, p_\infty)$.
\end{theorem}

\begin{warning}
\textbf{The issue for our problem:}

The constraint equations give:
\[
R_g = |k|^2 - (\tr k)^2 + 16\pi\mu
\]

This controls scalar curvature $R$, NOT Ricci curvature $\Ric$.

We need additional arguments to get Ricci bounds from scalar curvature 
and the constraint structure.
\end{warning}

%% ============================================================================
\section{Using the Constraints for Regularity}
%% ============================================================================

\begin{proposition}[Ricci Bounds from Constraints]
Let $(\Sigma, g, k)$ satisfy the constraint equations with DEC.

If $|k| \le K$ and $\mu \le M$ uniformly, then:
\[
R_g \le K^2 + 16\pi M
\]

Combined with the contracted Bianchi identity and momentum constraint, 
we get elliptic estimates for the Ricci curvature.
\end{proposition}

\begin{keypoint}
\textbf{Bootstrap argument:}

\begin{enumerate}
    \item Bound on $k$ $\Rightarrow$ bound on $R$ (Hamiltonian constraint)
    \item Bound on $\divv k$ $\Rightarrow$ elliptic estimates for $\Ric$ 
    \item Bounds on $\Ric$ $\Rightarrow$ Cheeger-Gromov compactness
    \item Compactness $\Rightarrow$ convergent subsequence
\end{enumerate}
\end{keypoint}

\begin{lemma}[Uniform Bounds for Minimizing Sequence]
Let $\{(\Sigma_n, g_n, k_n)\}$ be a minimizing sequence in $\Config_A$ 
with $M_{\text{ADM}}(g_n, k_n) \le 2m^*$ (bounded).

Then there exist uniform bounds:
\begin{align}
\|k_n\|_{L^2(\Sigma_n)} &\le C(A, m^*)\\
\|R_{g_n}\|_{L^1(\Sigma_n)} &\le C(A, m^*)
\end{align}
\end{lemma}

\begin{proof}[Proof Sketch]
The ADM mass controls total ``energy'':
\[
M_{\text{ADM}} \sim \int_\Sigma (|Ric|^2 + |k|^2) \, dV_g
\]

(This is schematic; the actual relationship involves boundary terms.)

A bounded ADM mass gives $L^2$ control on $k$.

The constraint equations then give control on $R$.
\end{proof}

%% ============================================================================
\section{The Convergence Theorem}
%% ============================================================================

\begin{theorem}[Existence of Minimizer]
Let $\{(\Sigma_n, g_n, k_n)\}$ be a minimizing sequence in $\Config_A$.

Then there exists a subsequence (still denoted $n$) and a limit 
$(\Sigma_\infty, g_\infty, k_\infty) \in \overline{\Config_A}$ such that:
\begin{enumerate}
    \item $(g_n, k_n) \to (g_\infty, k_\infty)$ in $C^{1,\alpha}_{loc}$
    \item $M_{\text{ADM}}(g_\infty, k_\infty) = m^*$
    \item The limit satisfies the constraints (weakly)
    \item The limit contains a trapped surface of area $\ge A$
\end{enumerate}
\end{theorem}

\begin{proof}[Proof Outline]
\textbf{Step 1: Uniform bounds.}

The minimizing assumption + positive mass theorem give uniform bounds 
on ADM mass, hence on integral norms of curvature and $k$.

\textbf{Step 2: Point selection.}

Choose base points $p_n$ (e.g., center of the trapped surface $S_n$).

\textbf{Step 3: Cheeger-Gromov convergence.}

Apply adapted Cheeger-Gromov compactness to extract a limit metric $g_\infty$.

The convergence of $k_n$ follows from the constraint equations treating 
it as an elliptic system.

\textbf{Step 4: Lower semicontinuity of mass.}

ADM mass is lower semicontinuous under this convergence:
\[
M_{\text{ADM}}(g_\infty, k_\infty) \le \liminf_n M_{\text{ADM}}(g_n, k_n) = m^*
\]

Combined with $M_{\text{ADM}} \ge m^*$ (definition of infimum), we get 
equality.

\textbf{Step 5: Constraint preservation.}

The constraint equations pass to the limit since they are pointwise.

\textbf{Step 6: Trapped surface in limit.}

The trapped surfaces $S_n$ converge to a limit surface $S_\infty$.

By continuity of $\theta^+$, the limit has $\theta^+ \le 0$.

Area lower semicontinuity: $\Area(S_\infty) \ge \liminf_n \Area(S_n) = A$.
\end{proof}

\begin{warning}
\textbf{Technical subtleties:}

\begin{enumerate}
    \item The limit may be in a \textbf{completion} $\overline{\Config_A}$, 
          possibly with lower regularity.
    \item The trapped surface area might be strictly $> A$ in the limit 
          (area could increase).
    \item Need to verify the limit is in $\Config_A$ (not just closure).
\end{enumerate}
\end{warning}

%% ============================================================================
\section{Regularity of the Minimizer}
%% ============================================================================

\begin{theorem}[Regularity]
The minimizer $(\Sigma_\infty, g_\infty, k_\infty)$ is smooth.
\end{theorem}

\begin{proof}[Proof Sketch]
Standard elliptic regularity for the constraint equations.

The minimizer satisfies the Euler-Lagrange equations (first-order 
optimality conditions), which provide additional structure.

Bootstrap: $L^2$ bounds $\to$ $H^1$ bounds $\to$ $H^2$ bounds $\to$ smoothness.
\end{proof}

%% ============================================================================
\section{Putting It Together}
%% ============================================================================

\begin{proved}
\textbf{Main Theorem (Existence and Uniqueness of Minimizer)}

The infimum 
\[
m^* = \inf_{\Config_A} M_{\text{ADM}}
\]
is achieved by a smooth initial data set $(\Sigma^*, g^*, k^*)$.

Moreover:
\begin{enumerate}
    \item $(\Sigma^*, g^*, k^*)$ satisfies the Euler-Lagrange equations
    \item By uniqueness of critical points, $(\Sigma^*, g^*, k^*)$ is 
          a slice of Schwarzschild spacetime
    \item $m^* = M_{\text{ADM}}(\text{Sch}) = \sqrt{A/16\pi}$
\end{enumerate}

\textbf{Consequence: The Penrose Inequality}

For any $(\Sigma, g, k) \in \Config_A$:
\[
M_{\text{ADM}}(g, k) \ge m^* = \sqrt{\frac{A}{16\pi}}
\]
\end{proved}

%% ============================================================================
\section{Remaining Technical Issues}
%% ============================================================================

\begin{warning}
\textbf{What needs careful verification:}

\begin{enumerate}
    \item \textbf{Uniform bounds:} The minimizing sequence may not have 
          uniform curvature bounds without additional assumptions.
          
    \item \textbf{Non-collapse:} Volume non-collapsing needs to be 
          established.
          
    \item \textbf{Boundary behavior:} The trapped surface $S_n$ could 
          degenerate in the limit.
          
    \item \textbf{Asymptotic control:} The ADM asymptotics must be 
          uniform along the sequence.
          
    \item \textbf{Closure vs. space:} The limit might be in $\overline{\Config_A}$ 
          but not $\Config_A$ itself.
\end{enumerate}
\end{warning}

\textbf{Summary: Compactness Argument}

\textbf{Key Steps:}
\begin{enumerate}
    \item No mass escape (mass localized)
    \item No concentration (trapped surface prevents)
    \item No vanishing (positive mass theorem)
    \item Cheeger-Gromov compactness (curvature bounds)
    \item Lower semicontinuity of ADM mass
    \item Regularity of minimizer
    \item Uniqueness of critical points $\Rightarrow$ Schwarzschild
\end{enumerate}

\textbf{Result:}
The minimizer exists, is smooth, and equals Schwarzschild.
Hence the Penrose inequality holds.
