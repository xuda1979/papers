\begin{abstract}
We perform a detailed calculation to determine whether the product Geroch
functional $\mathcal{G}_P = \sqrt{A/(16\pi)} \exp(-\frac{1}{16\pi}\int \theta^+\theta^-)$
is monotonic under the product expansion flow.
\end{abstract}

%===========================================================================
\section{Setup}
%===========================================================================

\subsection{Definitions}

We have initial data $(M, g, k)$ satisfying the dominant energy condition (DEC):
\begin{equation}
    \mu \geq |J|
\end{equation}
where $\mu = \frac{1}{2}(R + (\tr k)^2 - |k|^2)$ and $J_i = \nabla^j(k_{ij} - (\tr k)g_{ij})$.

For a surface $\Sigma$ with outward normal $\nu$:
\begin{align}
    H &= \text{mean curvature (trace of second fundamental form of $\Sigma$ in $M$)} \\
    \tr_\Sigma k &= g^{ab}k_{ab} \text{ restricted to $\Sigma$ (NOT $k_{ij}\nu^i\nu^j$)} \\
    \theta^+ &= H + \tr_\Sigma k \\
    \theta^- &= H - \tr_\Sigma k \\
    \mathcal{P} &= \theta^+\theta^- = H^2 - (\tr_\Sigma k)^2
\end{align}

The Product Expansion Flow:
\begin{equation}
    \frac{\partial \Sigma}{\partial t} = -\mathcal{H}_P \nu = -\sqrt{|\theta^+\theta^-|} \nu
\end{equation}

For trapped surfaces: $\theta^+ \leq 0$, $\theta^- < 0$, so $\mathcal{H}_P = \sqrt{\theta^+\theta^-} \geq 0$.

%===========================================================================
\section{Evolution Equations}
%===========================================================================

\subsection{Area Evolution}

For a flow $\partial\Sigma/\partial t = f\nu$:
\begin{equation}
    \frac{dA}{dt} = \int_\Sigma Hf \, dA
\end{equation}

With $f = -\mathcal{H}_P$:
\begin{equation}
    \boxed{\frac{dA}{dt} = -\int_\Sigma H\mathcal{H}_P \, dA}
\end{equation}

For trapped surfaces, $H < 0$ and $\mathcal{H}_P \geq 0$, so $\frac{dA}{dt} \geq 0$. \checkmark

\subsection{Evolution of Mean Curvature $H$}

Standard formula for $\partial\Sigma/\partial t = f\nu$:
\begin{equation}
    \frac{dH}{dt} = -\Delta_\Sigma f - (|A|^2 + \Ric(\nu,\nu))f
\end{equation}
where $|A|^2$ is the squared norm of the second fundamental form.

With $f = -\mathcal{H}_P$:
\begin{equation}
    \boxed{\frac{dH}{dt} = \Delta_\Sigma \mathcal{H}_P + (|A|^2 + \Ric(\nu,\nu))\mathcal{H}_P}
\end{equation}

\subsection{Evolution of $\tr_\Sigma k$}

This is more complicated. For a flow $\partial\Sigma/\partial t = f\nu$:
\begin{align}
    \frac{d(\tr_\Sigma k)}{dt} &= f\cdot(\nu^i\nu^j\nabla_i k_{jl}\nu^l + \text{terms involving $A$ and $k$})
\end{align}

The full formula involves:
\begin{equation}
    \frac{d(\tr_\Sigma k)}{dt} = f\cdot\left[\nabla_\nu(\tr_\Sigma k) + k(\nu, \nabla H) + 2A \cdot k^\top + \text{lower order}\right]
\end{equation}

This is quite involved. Let me use a cleaner form.

\begin{lemma}[Evolution of $\tr_\Sigma k$ - Correct Form]
\begin{equation}
    \frac{d(\tr_\Sigma k)}{dt} = f\cdot\left[\nabla_\nu(\tr_\Sigma k) - 2k(A) + \tr_\Sigma(\nabla_\nu k)\right]
\end{equation}
where $k(A)$ denotes the contraction $k_{ab}A^{ab}$.
\end{lemma}

\subsection{Evolution of $\mathcal{P} = H^2 - (\tr_\Sigma k)^2$}

\begin{equation}
    \frac{d\mathcal{P}}{dt} = 2H\frac{dH}{dt} - 2(\tr_\Sigma k)\frac{d(\tr_\Sigma k)}{dt}
\end{equation}

Substituting:
\begin{align}
    \frac{d\mathcal{P}}{dt} &= 2H\left[\Delta_\Sigma\mathcal{H}_P + (|A|^2 + \Ric(\nu,\nu))\mathcal{H}_P\right] \\
    &\quad - 2(\tr_\Sigma k)\cdot(-\mathcal{H}_P)\left[\text{spacetime terms}\right]
\end{align}

%===========================================================================
\section{The Critical Calculation}
%===========================================================================

\subsection{Computing $d\mathcal{G}_P/dt$}

Define:
\begin{equation}
    \mathcal{G}_P = \sqrt{\frac{A}{16\pi}} \exp\left(-\frac{1}{16\pi}\int_\Sigma \mathcal{P} \, dA\right)
\end{equation}

Take logarithm:
\begin{equation}
    \log\mathcal{G}_P = \frac{1}{2}\log\frac{A}{16\pi} - \frac{1}{16\pi}\int_\Sigma \mathcal{P} \, dA
\end{equation}

Differentiate:
\begin{equation}
    \frac{d}{dt}\log\mathcal{G}_P = \frac{1}{2A}\frac{dA}{dt} - \frac{1}{16\pi}\frac{d}{dt}\int_\Sigma \mathcal{P} \, dA
\end{equation}

For the second term:
\begin{equation}
    \frac{d}{dt}\int_\Sigma \mathcal{P} \, dA = \int_\Sigma \left(\frac{d\mathcal{P}}{dt} + \mathcal{P} H f\right) dA = \int_\Sigma \left(\frac{d\mathcal{P}}{dt} - \mathcal{P} H \mathcal{H}_P\right) dA
\end{equation}

\subsection{First Term}

\begin{equation}
    \frac{1}{2A}\frac{dA}{dt} = -\frac{1}{2A}\int_\Sigma H\mathcal{H}_P \, dA = -\frac{1}{2}\langle H\mathcal{H}_P\rangle
\end{equation}
where $\langle \cdot \rangle$ denotes the average over $\Sigma$.

\subsection{Second Term}

This requires the full evolution $d\mathcal{P}/dt$.

\textbf{Key observation:} Unlike the standard Geroch monotonicity (which uses IMCF $f = 1/H$), our flow has $f = -\mathcal{H}_P$.

For IMCF, the miraculous cancellation occurs because $\frac{dH}{dt} = \frac{\Delta H}{H} + \frac{|\nabla H|^2}{H^2} + (|A|^2 + \Ric)\frac{1}{H}$.

For our flow, we get different terms that don't simplify as nicely.

%===========================================================================
\section{The Obstruction}
%===========================================================================

\begin{tcolorbox}[colback=red!5, colframe=red!75!black, title=Critical Issue]
The Product Expansion Flow does \textbf{not} have the same miraculous
cancellation properties as IMCF.

Specifically:
\begin{itemize}
    \item IMCF: The speed $1/H$ is precisely tuned to make the Geroch functional monotonic.
    \item Product flow: The speed $\sqrt{|\theta^+\theta^-|}$ does not have this property.
\end{itemize}
\end{tcolorbox}

\subsection{Detailed Comparison}

For the \textbf{standard Hawking mass} under IMCF ($f = 1/H$):
\begin{equation}
    \frac{dm_H}{dt} = \sqrt{\frac{A}{16\pi}}\cdot\frac{1}{8\pi}\int_\Sigma\left[\frac{|\nabla H|^2}{H^3} + \frac{(\Ric(\nu,\nu) + \frac{1}{2}|A^\circ|^2)}{H}\right]dA \geq 0
\end{equation}
(assuming $H < 0$ and appropriate curvature conditions)

For the \textbf{product mass} under product flow ($f = -\mathcal{H}_P$):

The calculation does not simplify to give a manifestly non-negative quantity.

\subsection{Attempting a Different Functional}

Perhaps a different functional would work?

\begin{definition}[Alternative Product Functional]
\begin{equation}
    \tilde{\mathcal{G}}_P := \int_\Sigma \frac{1}{\mathcal{H}_P} \, dA \quad \text{(when $\mathcal{H}_P > 0$)}
\end{equation}
\end{definition}

Under the product flow:
\begin{equation}
    \frac{d}{dt}\int \frac{1}{\mathcal{H}_P}dA = \int\left[-\frac{1}{\mathcal{H}_P^2}\frac{d\mathcal{H}_P}{dt} + \frac{H(-\mathcal{H}_P)}{\mathcal{H}_P}\right]dA
\end{equation}

This also doesn't simplify nicely.

%===========================================================================
\section{Fundamental Problem}
%===========================================================================

\begin{tcolorbox}[colback=yellow!10, colframe=yellow!50!black, title=The Core Issue]
\textbf{Why IMCF works for Riemannian Penrose:}
\begin{itemize}
    \item The speed $1/H$ is the \emph{unique} choice that makes area increase at a controlled rate
    \item The Geroch functional $\sqrt{A}(1 - \frac{1}{16\pi}\int H^2)$ has special structure
    \item The DEC in Riemannian case simplifies to $R \geq 0$, which gives $\Ric(\nu,\nu) \geq 0$ contribution
\end{itemize}

\textbf{Why the product flow doesn't work:}
\begin{itemize}
    \item $\mathcal{H}_P = \sqrt{|\theta^+\theta^-|}$ is not tuned to any particular functional
    \item The ``product Geroch'' functional is ad hoc
    \item No cancellation between $H$ and $\tr_\Sigma k$ terms
\end{itemize}
\end{tcolorbox}

\subsection{What Would Be Needed}

For a Geroch-type monotonicity formula, we need a flow speed $f$ and functional $\mathcal{F}$ such that:
\begin{equation}
    \frac{d\mathcal{F}}{dt} = \int_\Sigma (\text{DEC-controlled terms}) \cdot f \, dA \geq 0
\end{equation}

The challenge: Find $f$ and $\mathcal{F}$ that:
\begin{enumerate}
    \item Don't depend on the sign of $\tr_\Sigma k$
    \item Give monotonicity under DEC
    \item Have the right limit at infinity and at trapped surfaces
\end{enumerate}

This is an \textbf{overdetermined system}---it's not clear such $(f, \mathcal{F})$ exists.

%===========================================================================
\section{Conclusion}
%===========================================================================

\begin{tcolorbox}[colback=red!5, colframe=red!75!black, title=Gap Confirmed]
\textbf{The Product Expansion Method has a fatal gap:}

The proposed monotonicity for $\mathcal{G}_P$ under the product flow is
\textbf{not established}. The calculation does not give a manifestly
non-negative quantity.

The key issue is that $\mathcal{H}_P = \sqrt{|\theta^+\theta^-|}$ is not
the correct flow speed for any known Geroch-type functional.
\end{tcolorbox}

\begin{tcolorbox}[colback=blue!5, colframe=blue!75!black, title=What Remains Valuable]
However, the \textbf{sign-invariance} of $\theta^+\theta^-$ is still a
valuable observation:
\begin{itemize}
    \item It avoids the conformal transformation obstruction
    \item It suggests looking for new functionals that use $\theta^+\theta^-$
    \item The area monotonicity under product flow is valid (just not the mass)
\end{itemize}

\textbf{Future direction:} Find the correct $(f, \mathcal{F})$ pair that
exploits $\theta^+\theta^-$ sign-invariance while achieving mass monotonicity.
\end{tcolorbox}
