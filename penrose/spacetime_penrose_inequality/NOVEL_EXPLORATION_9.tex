\section{Motivation}

Modern geometry goes beyond smooth manifolds:
\begin{itemize}
    \item Alexandrov spaces (curvature via triangle comparison)
    \item RCD spaces (Ricci curvature for metric-measure spaces)
    \item Lorentzian length spaces
\end{itemize}

These ``synthetic'' approaches might offer new perspectives on Penrose.

\section{Alexandrov Geometry}

\subsection{Triangle Comparison}

A metric space $(X, d)$ has curvature $\geq \kappa$ if triangles are ``fatter'' than in model space $M_\kappa$.

\begin{definition}
$(X, d)$ is $\text{CAT}(\kappa)$ (non-positive curvature) if triangles are ``thinner'' than in $M_\kappa$.
\end{definition}

\subsection{Application to GR?}

In Lorentzian geometry, ``timelike triangles'' make sense.

\begin{observation}
Strong energy condition (SEC) corresponds to timelike Ricci $\geq 0$, which has synthetic analogues.
\end{observation}

But DEC is different from SEC!

\subsection{The Trapped Triangle Test}

\begin{definition}
A \textbf{trapped triangle test}: 

For three points $p, q, r$ on a trapped surface, compare the triangle $pqr$ to a triangle in a model space.
\end{definition}

If the geometry is ``more curved'' than Schwarzschild, areas should be bounded.

\section{RCD Spaces}

\subsection{Definition}

$\text{RCD}(K, N)$ spaces have:
\begin{itemize}
    \item Synthetic Ricci curvature $\geq K$
    \item Dimension $\leq N$
\end{itemize}

\subsection{Properties}

On $\text{RCD}(K, N)$ spaces:
\begin{itemize}
    \item Bishop-Gromov volume comparison
    \item Splitting theorem (for $K = 0$)
    \item Bochner inequality
\end{itemize}

\subsection{A Synthetic DEC?}

\begin{conjecture}
There exists a synthetic formulation of DEC:
\[
    \text{``} \mu \geq |J| \text{ in synthetic sense''}
\]
that makes sense on non-smooth spaces.
\end{conjecture}

Key objects:
\begin{itemize}
    \item Timelike distance function $\tau(p, q)$
    \item Causal structure
    \item Synthetic curvature bounds
\end{itemize}

\subsection{Trapped Surfaces in This Framework}

\begin{definition}
In a Lorentzian length space, a surface $\Sigma$ is \textbf{synthetically trapped} if:
\[
    \tau(\Sigma, J^+(\Sigma)) = 0
\]
i.e., the future is ``immediately cut off.''
\end{definition}

\subsection{Synthetic Penrose?}

\begin{conjecture}[Synthetic Penrose]
In a Lorentzian length space with synthetic DEC:
\[
    M_{\ADM} \geq \sqrt{\frac{\mathcal{H}^2(\Sigma)}{16\pi}}
\]
where $\mathcal{H}^2$ is the 2-dimensional Hausdorff measure.
\end{conjecture}

\section{Metric Measure Spaces}

\subsection{Definition}

A metric-measure space is $(X, d, \mathfrak{m})$:
\begin{itemize}
    \item $(X, d)$: metric space
    \item $\mathfrak{m}$: reference measure
\end{itemize}

\subsection{The Mass Measure}

\begin{definition}
The \textbf{synthetic mass measure} on initial data is:
\[
    d\mathfrak{m}_M = \mu \, dV
\]
where $\mu$ is the energy density from DEC.
\end{definition}

\subsection{Total Mass}

\[
    \mathfrak{m}_M(M) = \int_M \mu \, dV
\]

Under what conditions does $\mathfrak{m}_M(M) = M_{\ADM}$?

\section{Comparison Geometry}

\subsection{Volume Comparison}

Bishop-Gromov: If $\text{Ric} \geq (n-1)K$, then:
\[
    \frac{\text{Vol}(B_r(p))}{\text{Vol}_{M_K}(B_r)}
\]
is non-increasing in $r$.

\subsection{Area Comparison?}

For surfaces:
\begin{conjecture}
If $(M, g)$ satisfies DEC and is asymptotically flat, then for a trapped surface $\Sigma$:
\[
    \frac{\Area(\Sigma)}{\Area_{S_{\text{Sch}}}(\text{corresponding surface in Schwarzschild})} \leq 1
\]
\end{conjecture}

The ``corresponding surface'' needs definition!

\subsection{The Schwarzschild Model}

In Schwarzschild of mass $M$:
\begin{itemize}
    \item Horizon at $r = 2M$ has $A = 16\pi M^2$
    \item Inner surface at $r < 2M$ has $A = 4\pi r^2$
\end{itemize}

\begin{conjecture}[Schwarzschild Comparison]
For any trapped surface $\Sigma$ in $(M, g, k)$ satisfying DEC:
\[
    \Area(\Sigma) \leq \text{Area of surface with same } m_H \text{ in Schwarzschild}
\]
\end{conjecture}

\section{Gromov-Hausdorff Convergence}

\subsection{Definition}

$(X_n, d_n) \xrightarrow{GH} (X, d)$ if there exist approximate isometries.

\subsection{Stability of Penrose}

\begin{conjecture}[Stability]
If $(M_n, g_n, k_n) \xrightarrow{GH} (M, g, k)$ with:
\begin{enumerate}
    \item Each $(M_n, g_n, k_n)$ satisfies Penrose
    \item DEC is preserved in limit
\end{enumerate}
Then $(M, g, k)$ satisfies Penrose.
\end{conjecture}

This would allow proving Penrose by approximation!

\subsection{Approximation Strategy}

\begin{enumerate}
    \item Prove Penrose for ``nice'' spaces (smooth, small $|\tr_\Sigma k|$)
    \item Approximate general spaces by nice ones
    \item Use stability
\end{enumerate}

\section{Non-Smooth Initial Data}

\subsection{Corners and Edges}

What if the trapped surface has corners?

The mean curvature $H$ has distributional part.

\subsection{Generalized Hawking Mass}

\begin{definition}
For a polyhedral surface $\Sigma$:
\[
    m_H^{gen}(\Sigma) = \sqrt{\frac{\Area(\Sigma)}{16\pi}}\left(1 - \frac{1}{16\pi}\int_{\text{smooth}} H^2 - \frac{1}{16\pi}\sum_{\text{edges}} \alpha_e L_e\right)
\]
where $\alpha_e$ is the dihedral angle deficit and $L_e$ is edge length.
\end{definition}

\subsection{Penrose for Polyhedral?}

\begin{conjecture}
Penrose holds for polyhedral trapped surfaces:
\[
    M_{\ADM} \geq \sqrt{\frac{\Area(\Sigma)}{16\pi}}
\]
even when $\Sigma$ is not smooth.
\end{conjecture}

\section{The Distance Function Approach}

\subsection{Distance to Infinity}

Define $r(x) = d(x, \text{infinity})$ (regularized).

In asymptotically flat space: $r(x) \to \infty$ as $x \to \infty$.

\subsection{Level Sets}

The level sets $\{r = c\}$ are surfaces.

\begin{definition}
The \textbf{distance sphere} at level $c$ is:
\[
    S_c = \{x : r(x) = c\}
\]
\end{definition}

\subsection{Area Growth}

\[
    \Area(S_c) \sim 4\pi c^2 \quad \text{as } c \to \infty
\]

The deviation from Euclidean encodes mass:
\[
    \Area(S_c) = 4\pi c^2 - 8\pi M c + O(1)
\]

So: $M = \lim_{c\to\infty} \frac{4\pi c^2 - \Area(S_c)}{8\pi c}$.

\subsection{Area of Trapped Surface}

\begin{conjecture}
For a trapped surface $\Sigma$ at ``distance'' $c_0$ from infinity:
\[
    \Area(\Sigma) \leq 4\pi c_0^2 - 8\pi M c_0 + O(1)
\]
which for $c_0 = 2M$ gives $\Area \leq 16\pi M^2 - 16\pi M^2 + O(1)$.
\end{conjecture}

The scaling is right but needs refinement.

\section{Optimal Transport on Metric Spaces}

\subsection{Wasserstein Distance}

On metric-measure spaces:
\[
    W_p(\mu, \nu) = \left(\inf_\gamma \int d(x,y)^p \, d\gamma\right)^{1/p}
\]

\subsection{Transport and Curvature}

Curvature bounds are equivalent to:
\begin{itemize}
    \item Convexity of entropy along geodesics
    \item Contraction of Wasserstein distance
\end{itemize}

\begin{conjecture}
DEC implies some contraction property for transport on initial data.
\end{conjecture}

\section{Conclusion}

\begin{tcolorbox}[colback=blue!10, colframe=blue!75!black]
\textbf{Synthetic Geometry Insights:}

\begin{enumerate}
    \item Synthetic curvature bounds (Alexandrov, RCD) might generalize DEC
    \item Lorentzian length spaces give non-smooth causal structure
    \item Gromov-Hausdorff stability could enable approximation proofs
    \item Distance function encodes ADM mass in area growth
    \item Comparison geometry might give Schwarzschild as extremal case
\end{enumerate}

\textbf{Most Promising:}

The \textbf{Schwarzschild comparison} approach:
\begin{itemize}
    \item Schwarzschild saturates Penrose
    \item DEC might imply ``less curved than Schwarzschild'' for trapped surfaces
    \item Area comparison would give Penrose
\end{itemize}

\textbf{Challenge:}

Making ``comparison to Schwarzschild'' precise when the topology and data are different.
\end{tcolorbox}
